    <main>
        <!-- Inline CSS Example -->
        <div style="background-color: #ffebee; border: 2px dashed #f44336; 
                    padding: 15px; margin: 20px 0; border-radius: 5px;">
            <h2 style="color: #c62828; margin-top: 0;">Inline CSS Styling</h2>
            <p style="color: #424242; line-height: 1.5;">
                This div is styled using <strong>inline CSS</strong> with the style attribute directly in the HTML elements.
                Inline styles have the highest specificity and will override other styles.
            </p>
            <ul style="list-style-type: square; color: #616161;">
                <li>Highest specificity</li>
                <li>Hard to maintain</li>
                <li>Use sparingly for one-off styles</li>
            </ul>
        </div>
        
        <!-- Internal CSS Example -->
        <div class="internal-style">
            <h2>Internal CSS Styling</h2>
            <p>
                This div is styled using <strong>internal CSS</strong> with style tags in the head section.
                Internal CSS is better for single-page styling as it keeps styles separate from HTML structure.
            </p>
            <ul>
                <li>Medium specificity</li>
                <li>Better organization than inline</li>
                <li>Good for single-page applications</li>
            </ul>
        </div>
        
        <!-- External CSS Example -->
        <div class="external-style">
            <h2>External CSS Styling</h2>
            <p>
                This div is styled using <strong>external CSS</strong> from a separate styles.css file.
                External CSS is the most maintainable approach for multi-page websites.
            </p>
            <ul>
                <li>Lowest specificity (but most maintainable)</li>
                <li>Best for large projects</li>
                <li>Enables style reuse across pages</li>
            </ul>
        </div>
        
        <!-- Specificity Demonstration -->
